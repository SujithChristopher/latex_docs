    h_1(0) V_b = \frac{1}{r_i} \left[\begin{array}{cc}
            1 & \tan\gamma_i \\

            \cos \beta_i    & \sin \beta_i \\
            \- \sin \beta_i & \cos \beta_i \\
        \end{array}\right] V_b

    \gamma_1 = 0, \beta_1 = 0
\end{equation}
